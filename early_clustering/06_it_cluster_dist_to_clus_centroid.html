clus_bdg <- function(sfdf, scaled, fd, sd, mda = 0, mds = 0) {
    
   if(nrow(mda) & nrow(mds)) {
    adm <- mda
    sdm <- mds
   } else {
        # Prparation et sparation des donnes (non-)spatiales
    bgs_1 <- sfdf %>%
                remove_rownames() %>%
                column_to_rownames('id')
    bgs_data_1 <- bgs_1 %>%
                    st_set_geometry(NULL)
    
    # Calcul des matrices de distance
    adm <- if(scaled) {
                as.data.frame(as.matrix(dist(bgs_data_1[, c(-31:-34, -36:-39)]))) #### !!!!!! A CHANGER PAR LA SUITE
            } else {
                as.data.frame(as.matrix(dist(scale(bgs_data_1[, c(-31:-34, -36:-39)]))))
            }
    sdm <- st_distance(bgs_1)
    rownames(sdm) <- colnames(sdm) <- rownames(adm)
   }
    
    # Prparation df rsultats
    final_clus <- as.data.frame(cbind(rownames(adm), integer(length(nrow(sdm)))), stringsAsFactors = F)
        colnames(final_clus) <- c('id', 'clus') # Rename col
    
        # Return a df with all bdgs still to class
        remaining_rows <- function(df_clus) {
    
        r_rows <- df_clus %>%
                    filter(clus == 0)
        return(r_rows)
    }
    
    # Retrieve the first round of neighbours for the given building
    find_first_nbrs <- function(bat_id, attdm, spadm) {
    
        df_bat <- as.data.frame(cbind(adm[,bat_id], sdm[,bat_id])) # Distances to every bdg
        colnames(df_bat) <- c('dist_att', 'dist_spat') # Rename col
        df_bat$id <- rownames(sdm) # Add a col w/ ids
        near_bdgs_id <- df_bat %>% 
                            filter(dist_att <= fd & dist_spat <= sd & id != bat_id) %>% # Bds under threshold, except the origin
                            pull(id) # return the column as a vector
    
        return(near_bdgs_id)
    }

    find_nbrs_of_nbrs <- function(nbrs_id) {
        n_id <- character() # Create empty vector for ids
    
        for(bdg in nbrs_id) {
            df_bat_n <- as.data.frame(cbind(adm[,bdg], sdm[,bdg])) # Distances to every bdg
            colnames(df_bat_n) <- c('dist_att', 'dist_spat') # Rename col
            df_bat_n$id <- rownames(sdm) # Add a rowname column
            near_bdgs_n <- df_bat_n %>% 
                               filter(dist_att <= fd & dist_spat <= sd & id != bdg)
            n_id <- c(n_id, near_bdgs_n$id) # Add new neighIDs to the vector of all id
        }
    
        return(unique(n_id)) # Return only the unique neighbours
    }
    
    cluster <- function(id_bd, cls_nb) {
        nbrs <- find_first_nbrs(id_bd, adm, sdm) # Trouve les voisins du premier id
        prev_lgt <- prev_lgt_nbrs <- 0 # Initialise les longueurs prcdentes  zro
        unique_ids <- append(id_bd, nbrs) # Cre le vecteur d'ids unique avec l'id donn et les premiers voisins
    
        for (i in 1:10000) {
            a <- find_nbrs_of_nbrs(nbrs) # Trouver les voisins des voisins
         
            if(!(length(a))) { # Si pas de voisin 
                final_clus[final_clus$id == id_bd, 'clus'] <<- cls_nb # Assigne l'id comme cluster
                break # Sort de la boucle
            
            } else if(length(a) == prev_lgt | length(unique(unique_ids)) == prev_lgt_nbrs) { # Si autant de voisins qu' l'itration prcdente
            
                for(j in unique(unique_ids)) { # Assigne aux voisins l'id du 1er batiment
                    final_clus[final_clus$id == j, 'clus'] <<- cls_nb
                }
                break # Sort de la boucle
            }
            prev_lgt_nbrs <- length(unique(unique_ids)) # Ajuste la longueur prcdente des ids uniques
            unique_ids <- append(unique_ids, a) # Ajoute les ids  la liste
        
            prev_lgt <- length(a) # Ajuste la longueur prcdente des voisins
            nbrs <- a # Assigne nbrs pour la prochaine itration
        }
    }
    
    for(i in final_clus$id) {
        if(i %in% remaining_rows(final_clus)$id) {
            cluster(i, i)
        }
    }
    
    final_clus$cls <- as.factor(as.integer(as.factor(final_clus$clus)))
    final_clus$clus <- NULL
    
    return(final_clus)

}

Distance spatiale entre clusters et enveloppes géométriques¶

test est un sfdf avec les attributs, gomtries et clusters d'attribution

Ainsi, pour chaque seuil, il est possible de connatre le centroide de l'enveloppe concave ($\alpha$-shape ou $\alpha$-hull) des clusters. Toutefois, plusieurs problmes se posent:

  • Les clusters avec peu de btiments $\Rightarrow$ utiliser les noeuds des polygones rsout le problme: Ne garder que les btiments du cluster slectionn, transformer les noeuds en points, extraire leur coordonnes et ne garder que les coordonnes uniques

    unique(st_coordinates(st_cast(filter(test, as.numeric(cls_3_100) == id_clus)$geom, 'POINT')))

  • La dfinition de l'$\alpha$, cela peut donner des rsultats assez diffrents, en fonction du nombre de btiments du cluster et de leur disposition. Il faut galement enregistrer, pour chaque cluster, le polygone correspondant l'$\alpha$-shape afin de l'utiliser (ou son centrode) ultrieurement.

ex_pol_from_ashape <- unique(st_coordinates(st_cast(filter(test, as.numeric(cls_3_100) == 108)$geom, 'POINT')))
ex_shape <- ashape(ex_pol_from_ashape, alpha = 100)

# Crer un graphe  partir des edges de l'ashape
tcirctest <- cbind(as.character(ex_shape$edges[, "ind1"]), as.character(ex_shape$edges[, 
    "ind2"]))
tcirc <- graph.edgelist(cbind(as.character(ex_shape$edges[, "ind1"]), as.character(ex_shape$edges[, 
    "ind2"])), directed = FALSE)
# Enlever une des edges du graphe
cutg = tcirc - E(tcirc)[1]
# Trouver le dernier noeud et crer la suite
ends = names(which(degree(cutg) == 1))
path = get.shortest.paths(cutg, ends[1], ends[2])[[1]]
# Index des points
pathX = as.numeric(names(unlist(path)))
# Rajout du premier point pour fermer le polygone
pathX = c(pathX, pathX[1])

heho <- NULL
for(i in pathX) {
    heho <- rbind(heho, ex_shape$x[i,])
}

p <- st_sfc(st_polygon(list(heho)), crs = 21781)

Coordonnées des centroïdes de polygones ou de l'enveloppe concave en cas de multipolygones¶

Cette fonction prend en entre le dataframe des attributs moyens des clusters et joint les coordonnes du centrode du polygone (1 btiment) ou de l'enveloppe concave (2 ou plus btiments) en sortie.

get_centroid <- function(cls_attr_df) {
    
    coords <- data.frame() # Un nouveau dataframe vide
    
    for(i in 1:nrow(cls_attr_df)) {
        row <- cls_attr_df[i, ] # Le cluster (rang) en cours
        
        if(st_geometry_type(row) == 'MULTIPOLYGON') { # S'il y a plusieurs btiments
            # Rcupre les coordonnes uniques des vertices des polygones de btiments
            ex_pol_from_ashape <- unique(st_coordinates(st_cast(st_sfc(row$geometry, crs = 21781), 'POINT', group_or_split = T)))
            # Cre l'alphashape sur la base des points prcdents. Ici alpha = 100
            ex_shape <- ashape(ex_pol_from_ashape, alpha = 100)
            
            # Crer un graphe  partir des edges de l'ashape
            tcirc <- graph.edgelist(cbind(as.character(ex_shape$edges[, "ind1"]), as.character(ex_shape$edges[, 
                "ind2"])), directed = FALSE)
            # Enlever une des edges du graphe         
            cutg = tcirc - E(tcirc)[1]
            # Trouver le dernier noeud et crer la suite
            ends = names(which(degree(cutg) == 1))
            path = get.shortest.paths(cutg, ends[1], ends[2])[[1]]
            # Index des points
            pathX = as.numeric(names(unlist(path)))
            # Rajout du premier point pour fermer le polygone
            pathX = c(pathX, pathX[1])
            
            coords_poly <- NULL
            
            for(i in pathX) { # Rcuprer les points dans le bon ordre
                coords_poly <- rbind(coords_poly, ex_shape$x[i,])
            }
            # Crer le polygone (ashape) sur la base des coordonnes rcupres
            p <- st_sfc(st_polygon(list(coords_poly)), crs = 21781)
            
            # Rajouter les coordonnes du centrodes (x et y) au df temporaire
            coords <- rbind(coords, st_coordinates(st_centroid(p)))

        } else {
            # Rajouter les coordonnes du centrode (x et y) du btiment au df temporaire
            coords <- rbind(coords, st_coordinates(st_centroid(row$geometry)))
        }
        
    }
    
    # Joindre les coordonnes x et y
    cd_centro <- apply(coords, 1, function(x) {
        
        paste(round(x[1], 2), round(x[2], 2), sep = ',')
    })
        
    return(cbind(cls_attr_df, cd_centro))
}
