plot_dists <- function(df_da, df_ds, v_cls, idx) {
    df_bat <- cbind(df_da[, c(idx, 2)], drop_units(df_ds[, idx])) # Bind the distances
    df_bat <- df_bat[, c(1,3)] # Keep only the correct distances (2 imported to keep the rownames)
    id <- colnames(df_bat)[1] # Retrieve the building ID
    df_bat <- cbind(df_bat, as.factor(v_cls)) # Bind the clusters
    colnames(df_bat) <- c("da", "ds", "cls") # Rename the columns
    
    # Plot the graph
    print(ggplot(df_bat, aes(ds, da, colour = cls)) + 
    geom_point() +
    scale_colour_brewer(type = "qual", palette = "Pastel1", aesthetics = "fill") +
    labs(title = paste0("Distances attributaires et spatiales, bt. n", id), 
         x = "Dist (m)", y = "Dist. euclid.", colour = "Clusters"))
}
